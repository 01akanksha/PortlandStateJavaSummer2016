11  /**
12   * Unit tests for the Student class.  In addition to the JUnit annotations,
13   * they also make use of the <a href="http://hamcrest.org/JavaHamcrest/">hamcrest</a>
14   * matchers for more readable assertion statements.
15   */
16  public class StudentTest
17  {
18  
19    @Test
20    public void studentNamedPatIsNamedPat() {
21      String name = "Pat";
22      Student pat = createStudentWithName(name);
23      assertThat(pat.getName(), equalTo(name));
24    }
25  
26    @Test
27    public void studentDescriptionContainsName() {
28      String name = "Pat";
29      Student student = createStudentWithName(name);
30      assertThat(student.toString(), containsString(name));
31    }
32  
33    private Student createStudentWithName(String name) {
34      return createStudentWithNameAndGpa(name, 0.0);
35    }
36  
37    @Test
38    public void studentDescriptionContainsGPA() {
39      double gpa = 1.23;
40      Student student = createStudentWithGPA(gpa);
41      assertThat(student.toString(), containsString(String.valueOf(gpa)));
42    }
43  
44    private Student createStudentWithGPA(double gpa) {
45      return createStudentWithNameAndGpa("Name", gpa);
46    }
47  
48    private Student createStudentWithNameAndGpa(String name, double gpa) {
49      return new Student(name, new ArrayList(), gpa, "Doesn't matter");
50    }
51  
52  }
