17   * A unit test for the {@link AppointmentBookServlet}.  It uses mockito to
18   * provide mock http requests and responses.
19   */
20  public class AppointmentBookServletTest {
21  
22    @Test
23    public void getOnServletPrettyPrintsPreCannedAppointmentBookOwner() throws ServletException, IOException {
24      AppointmentBookServlet servlet = new AppointmentBookServlet();
25  
26      HttpServletRequest request = mock(HttpServletRequest.class);
27      HttpServletResponse response = mock(HttpServletResponse.class);
28      PrintWriter pw = mock(PrintWriter.class);
29  
30      String ownerName = "PreCannedOwner";
31      when(request.getParameter("owner")).thenReturn(ownerName);
32  
33      when(response.getWriter()).thenReturn(pw);
34  
35      servlet.doGet(request, response);
36  
37      verify(pw).println(ownerName);
38      verify(response).setStatus(HttpServletResponse.SC_OK);
39    }
40  
41    @Test
42    public void postToServletCreatesAppointment() throws ServletException, IOException {
43      AppointmentBookServlet servlet = new AppointmentBookServlet();
44  
45      HttpServletRequest request = mock(HttpServletRequest.class);
46      HttpServletResponse response = mock(HttpServletResponse.class);
47      PrintWriter pw = mock(PrintWriter.class);
48  
49      String ownerName = "PreCannedOwner";
50      when(request.getParameter("owner")).thenReturn(ownerName);
51      String description = "My description";
52      when(request.getParameter("description")).thenReturn(description);
53      String beginTime = "1/1/2016 1:00 PM";
54      when(request.getParameter("beginTime")).thenReturn(beginTime);
55      String endTime = "1/2/2016 2:00 PM";
56      when(request.getParameter("endTime")).thenReturn(endTime);
57  
58  
59      when(response.getWriter()).thenReturn(pw);
60  
61      servlet.doPost(request, response);
62  
63      verify(response).setStatus(HttpServletResponse.SC_OK);
64  
65      AppointmentBook book = servlet.getAppointmentBookForOwner(ownerName);
66      Collection<Appointment> appointments = book.getAppointments();
67      assertThat(appointments.size(), equalTo(1));
68      Appointment appointment = appointments.iterator().next();
69      assertThat(appointment.getDescription(), equalTo(description));
70      assertThat(appointment.getBeginTimeString(), equalTo(beginTime));
71      assertThat(appointment.getEndTimeString(), equalTo(endTime));
72    }
